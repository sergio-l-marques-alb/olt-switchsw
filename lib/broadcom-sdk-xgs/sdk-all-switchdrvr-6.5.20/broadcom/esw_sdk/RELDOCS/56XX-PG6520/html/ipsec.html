IPSEC

bcm_ipsec_config_t Structure used for bcm_ipsec_config_set API.

typedef struct bcm_ipsec_config_s {
uint32 num_of_ipsec_channels;
uint32 num_of_bfd_channels;
} bcm_ipsec_config_t;

bcm_ipsec_tunnel_info_t Structure used for bcm_ipsec_tunnel_create API.

typedef struct bcm_ipsec_tunnel_info_s {
uint32 context_id; /* VRF ID */
bcm_ipv4_ipv6_addr_t local_ip; /* IPV4 or IPV6 */
bcm_ipv4_ipv6_addr_t remote_ip; /* IPV4 or IPV6 */
uint32 protocol; /* Tunnel protocol; IPV4 or IPV6 */
uint8 ttl;
uint8 dscp;
} bcm_ipsec_tunnel_info_t;
API Brief
bcm_ipsec_tunnel_create(int unit, uint32 flags, uint32* tunnel_id, bcm_ipsec_tunnel_info_t* tunnel_info) Add IPSEC tunnel
bcm_ipsec_tunnel_delete(int unit, uint32 flags, uint32 tunnel_id) Delete IPSEC tunnel

bcm_ipsec_direction_t Structure used for in bcm_ipsec_sa_info_t for bcm_ipsec_sa_create API.

bcm_ipsec_sa_info_t Structure used for bcm_ipsec_tunnel_add API.

typedef struct bcm_ipsec_sa_info_s {
uint32 initial_seq_num;
uint32 mtu;
uint8 fragmentation_enable;
bcm_ipsec_algo_type_t encrypt_type;
bcm_ipsec_auth_type_t auth_type;
bcm_ipsec_direction_t sa_direction;
uint32 spi;
} bcm_ipsec_sa_info_t;
API Brief
bcm_ipsec_sa_create(int unit, uint32 flags, uint32* sa_id, bcm_ipsec_sa_info_t* sa) Add IPSEC SA

bcm_ipsec_sa_keys_t Structure used for bcm_ipsec_sa_key_update API.

typedef struct bcm_ipsec_sa_keys_s {
